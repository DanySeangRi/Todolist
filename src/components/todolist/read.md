import React, { useState } from 'react' import './todolist.css' import { FaSlideshare } from "react-icons/fa"; import { BsPcDisplay } from "react-icons/bs"; import { FaRegComments } from "react-icons/fa6"; import { BiDotsHorizontal } from "react-icons/bi"; import { FaCirclePlus } from "react-icons/fa6"; import { MdDelete } from "react-icons/md"; const Feature = (props) => { const Icon = props.icon; const name = props.name; // vea yor domlai pi icon 1 1 return ( <div className='feature'> <button ><Icon size={20} /></button> <button>{name}</button> </div> ); }; const ToDoList = () => { const [task,setTask] =useState(['dada']); const [newTask,setNewTask] =useState(""); function handleInputChnage(event){ setNewTask(event.target.value) } function addTask(){ if(newTask.trim()!==""){ setTask(t=>[...t, { text: newTask, time: new Date() }]) setNewTask("") } } function deleteTask(index){ const updateTask = task.filter((_,i)=>i!==index) setTask(updateTask) } function deleteAll(){ setTask([]) } return ( <div className='todo-container'> <div className='todo-navbar'> <div> <p>My Project/</p> </div> <div className='todo-navbar__left'> <Feature icon={FaSlideshare} name="Search" /> <Feature icon={BsPcDisplay} name="Display" /> <Feature icon={FaRegComments} name="" /> <Feature icon={ BiDotsHorizontal} name="" /> </div> </div> <div className='todo-task'> <div className='add-task'> <h1>Today Task</h1> <div className='add-input '> <input type="text" name="" id="" placeholder='Enter something...' value={newTask} onChange ={handleInputChnage}/> <button className='add-input__btn-add' onClick={addTask}><FaCirclePlus size={20}/></button> <button className='add-input__btn-delete' onClick={deleteAll}>Delete All</button> </div> </div> <ol> { task.map((task,index)=> <div key={index} className='list'> <span>{task.text}</span> <span className="time"> {task.time.toLocaleTimeString([], { hour: "2-digit", minute: "2-digit" })} <button onClick={()=>deleteTask(index)}><MdDelete size={15} /></button> </span> </div> ) } </ol> </div> </div> ) } export default ToDoList
